Example Query Structure

-- Customer Segmentation by Account Value
WITH customer_segments AS (
SELECT
c.customer_id,
c.customer_name,
SUM(a.account_balance) as total_balance,
CASE
WHEN SUM(a.account_balance) >= 100000 THEN 'Premium'
WHEN SUM(a.account_balance) >= 50000 THEN 'Gold'
ELSE 'Silver'
END as segment,
ROW_NUMBER() OVER (PARTITION BY segment ORDER BY total_balance DESC) as rank
FROM customers c
LEFT JOIN accounts a ON c.customer_id = a.customer_id
GROUP BY c.customer_id, c.customer_name
)
SELECT * FROM customer_segments WHERE rank <= 10
ORDER BY segment, rank;
